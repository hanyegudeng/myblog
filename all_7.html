创建私有pods库

首先创建一个自己的git仓库。我创建的仓库是：http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git

1.pod repo add USeOnLineLib http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git
2.pod lib create USeOnLineLib
如果死在这里：Setting up CocoaPods master repo，退出，进到Example,update下pods就可以了。（出现这个原因是本地的pods被删除了）
3.推送创建的文件到git上

git add .
git commit -s -m "Initial Commit of Library"
git remote add origin http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git
git push origin master

4.因为pods必须要有tag所以创建一个tag

git tag -m "first release" 0.1.0
git push --tags

5.修改podspec文件。可以直接打开Example下的项目

  s.name             = 'USeOnLineLib'
  s.version          = '0.1.0'
  s.summary          = 'USeOnLineLib'

  s.description      = <<-DESC
TODO: Add long description of the pod here.
                       DESC

  s.homepage         = 'http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git'
  s.license          = { :type => 'MIT', :file => 'LICENSE' }
  s.author           = { 'Yun' => 'xxxxxx@163.com' }
  s.source           = { :git => 'http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git', :tag => '0.1.0' }

  s.ios.deployment_target = '8.0'

  s.source_files = 'USeOnLineLib/Classes/**/*'
  
  # s.resource_bundles = {
  #   'USeOnLineLib' => ['USeOnLineLib/Assets/*.png']
  # }

  # s.public_header_files = 'Pod/Classes/**/*.h'
  # s.frameworks = 'UIKit', 'MapKit'
  # s.dependency 'AFNetworking', '~> 2.3'
end

s.dependency 'AFNetworking', '~> 2.3' 这样就附件了AFNetworking库，多个就多行

6.验证podspec文件是否正确pod lib lint
如果出现下面信息，就说明验证成功


USeOnLineLib passed validation.

7.提交podspec（如果更新pod后，也可以使用该命令重新发布提交）

pod repo push USeOnLineLib USeOnLineLib.podspec

8.在项目中使用该私有pods。前提是使用者必须有该库的访问权限

pod 'USeOnLineLib',:git=>"http://xxx.xxx.xx.x:28080/Yun/USeOnLineLib.git"

在pods里面使用别的pods的类的时候导入的时候用import <>避免出问题

#!/usr/bin/python


import struct,socket
import hashlib
import threading,random
import time
import struct
from base64 import b64encode, b64decode


connectionlist = {}
g_code_length = 0
g_header_length = 0


def hex2dec(string_num):
    return str(int(string_num.upper(), 16))




def get_datalength(msg):
    global g_code_length
    global g_header_length

    print (len(msg))
    g_code_length = ord(msg[1]) & 127
    received_length = 0;
    if g_code_length == 126:
    #g_code_length = msg[2:4]
    #g_code_length = (ord(msg[2])<<8) + (ord(msg[3]))
        g_code_length = struct.unpack('>H', str(msg[2:4]))[0]

    elif g_code_length == 127:
        #g_code_length = msg[2:10]

def parse_data(msg):
    global g_code_length
    g_code_length = ord(msg[1]) & 127
    received_length = 0;
    if g_code_length == 126:
        g_code_length = struct.unpack('>H', str(msg[2:4]))[0]
        masks = msg[4:8]
        data = msg[8:]
    elif g_code_length == 127:
        g_code_length = struct.unpack('>Q', str(msg[2:10]))[0]
        masks = msg[10:14]
        data = msg[14:]
    else:
        masks = msg[2:6]
        data = msg[6:]


    i = 0
    raw_str = ''


    for d in data:
        raw_str += chr(ord(d) ^ ord(masks[i%4]))
        i += 1


    print (u"总长度是：%d" % int(g_code_length))
    return raw_str


def sendMessage(message):

def deleteconnection(item):
    global connectionlist
    del connectionlist['connection'+item]


class WebSocket(threading.Thread):#继承Thread


    GUID = "258EAFA5-E914-47DA-95CA-C5AB0DC85B11"


    def __init__(self,conn,index,name,remote, path="/"):
        threading.Thread.__init__(self)#初始化父类Thread
        self.conn = conn
        self.index = index
        self.name = name
        self.remote = remote
        self.path = path
        self.buffer = ""
        self.buffer_utf8 = ""
        self.length_buffer = 0
    def run(self):#重载Thread的run
        print('Socket%s Start!' % self.index)
        headers = {}
        self.handshaken = False


        while True:
            if self.handshaken == False:
                print ('Socket%s Start Handshaken with %s!' % (self.index,self.remote))
                self.buffer += bytes.decode(self.conn.recv(1024))


                if self.buffer.find('\r\n\r\n') != -1:
                    header, data = self.buffer.split('\r\n\r\n', 1)
                    for line in header.split("\r\n")[1:]:
                        key, value = line.split(": ", 1)
                        headers[key] = value


                    headers["Location"] = ("ws://%s%s" %(headers["Host"], self.path))
                    key = headers['Sec-WebSocket-Key']
                    token = b64encode(hashlib.sha1(str.encode(str(key + self.GUID))).digest())


                    handshake="HTTP/1.1 101 Switching Protocols\r\n"\
                        "Upgrade: websocket\r\n"\
                        "Connection: Upgrade\r\n"\
                        "Sec-WebSocket-Accept: "+bytes.decode(token)+"\r\n"\
                        "WebSocket-Origin: "+str(headers["Origin"])+"\r\n"\
                        "WebSocket-Location: "+str(headers["Location"])+"\r\n\r\n"


                    self.conn.send(str.encode(str(handshake)))
                    self.handshaken = True
                    print ('Socket %s Handshaken with %s success!' %(self.index, self.remote))
                    sendMessage(u'Welcome, ' + self.name + ' !')
                    self.buffer_utf8 = ""
                    g_code_length = 0


            else:
                global g_code_length
                global g_header_length
                mm=self.conn.recv(128)
                if len(mm) <= 0:
                    continue
                if g_code_length == 0:
                    get_datalength(mm)
                #接受的长度
                self.length_buffer = self.length_buffer + len(mm)
                self.buffer = self.buffer + mm
                if self.length_buffer - g_header_length < g_code_length :
                    continue
                else :
                    self.buffer_utf8 = parse_data(self.buffer) #utf8                
                    msg_unicode = str(self.buffer_utf8).decode('utf-8', 'ignore') #unicode
                    if msg_unicode=='quit':
                        print (u'Socket%s Logout!' % (self.index))
                        nowTime = time.strftime('%H:%M:%S',time.localtime(time.time()))
                        sendMessage(u'%s %s say: %s' % (nowTime, self.remote, self.name+' Logout'))
                        deleteconnection(str(self.index))
                        self.conn.close()
                        break #退出线程
                    else:
                        #print (u'Socket%s Got msg:%s from %s!' % (self.index, msg_unicode, self.remote))
                        nowTime = time.strftime(u'%H:%M:%S',time.localtime(time.time()))
                        sendMessage(u'%s %s say: %s' % (nowTime, self.remote, msg_unicode))
                    #重置buffer和bufferlength
                    self.buffer_utf8 = ""
                    self.buffer = ""
                    g_code_length = 0
                    self.length_buffer = 0
            self.buffer = ""


class WebSocketServer(object):
    def __init__(self):
        self.socket = None
    def begin(self):
        print( 'WebSocketServer Start!')
        self.socket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        self.socket.setsockopt(socket.SOL_SOCKET,socket.SO_REUSEADDR,1)
        self.socket.bind(("127.0.0.1",12345))
        self.socket.listen(50)


        global connectionlist


        i=0
        while True:
            connection, address = self.socket.accept()


            username=address[0]
            newSocket = WebSocket(connection,i,username,address)
            newSocket.start() #开始线程,执行run函数
            connectionlist['connection'+str(i)]=connection
            i = i + 1


if __name__ == "__main__":
    server = WebSocketServer()
    server.begin()

如何是用Targets来管理开发测试包

我们在开发APP的过程中难免会遇到开发测试分离的情况，最常见的就是API地址，是不同的，这个可以通过宏定义或者注释之类的来加以解决，但是打包的时候万一忘了就完蛋喽。其实xcode就能帮我们防呆，那就是创建多个Target来管理不同的环境，配置不同的plist以及对应的程序图标版本号等等的信息。具体做法如下：

1.选择原有的target，复制一份，名字可以修改，也可以不管
2.在Build Settings里面的Preprocessor Macros添加不同的DEVELOPMENT=0
3.在用到API host的地方加入判断

4.如果是用swift编写项目，需要在Swift Compiler - Custom Flags里面把DEVELOPMENT添加进去才可以起效
```

if DEVELOPMENT

define SERVER_URL @"http://dev.server.com/api/"

define API_TOKEN @"DI2023409jf90ew"

else

define SERVER_URL @"http://prod.server.com/api/"

define API_TOKEN @"71a629j0f090232"

endif


4.如果程序中使用cocoapods，并且使用了workspace，就要稍微配置下podfile文件了。

abstract_target 'targetBlock' do

platform :ios, "7.0"

xcodeproj 'xx/xx.xcodeproj'

pod 'AFNetworking', '~> 3.0'

target 'testing target'
target 'product target'

end
```

5.如果后续添加文件什么的，记得选择两个target哦。要不另外一个target就没有对应的文件喽

6.注意：
swift需要在Other Swift Flags里面添加
"-D" "DEVELOPMENT"
