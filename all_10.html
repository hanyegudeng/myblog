获取地图上下左右4个点坐标


//西北角坐标即左上角坐标
-(CLLocationCoordinate2D)getNWCoordinate{
    
    MKMapRect mRect = self.mapView.visibleMapRect;
    
    return [self getCoordinateFromMapRectanglePoint:MKMapRectGetMinX(mRect) y:mRect.origin.y];
}

//右上角

-(CLLocationCoordinate2D)getNECoordinate{
    
    MKMapRect mRect = self.mapView.visibleMapRect;
    
    return [self getCoordinateFromMapRectanglePoint:MKMapRectGetMaxX(mRect) y:mRect.origin.y];
}

//左下角

-(CLLocationCoordinate2D)getSWCoordinate{
    
    MKMapRect mRect = self.mapView.visibleMapRect;
    
    return [self getCoordinateFromMapRectanglePoint:MKMapRectGetMinX(mRect) y:MKMapRectGetMaxY(mRect)];
}

//右下角
-(CLLocationCoordinate2D)getSECoordinate{
    
    MKMapRect mRect = self.mapView.visibleMapRect;
    
    return [self getCoordinateFromMapRectanglePoint:MKMapRectGetMaxX(mRect) y:MKMapRectGetMaxY(mRect)];
}

-(CLLocationCoordinate2D)getCoordinateFromMapRectanglePoint:(double)x y:(double)y{
    MKMapPoint swMapPoint = MKMapPointMake(x, y);
    return MKCoordinateForMapPoint(swMapPoint);
}

+inf -inf NAN

+inf无限大，例如分母为0的时候，就会为无限大
-inf无限小，例如log(0)
NAN不是一个数字，例如sqrt（-1）
在oc里面如何判断呢
isinf(x)可以判断是否是无穷大
isnan(x)可以判断是否不是一个数字

NSURL *url = [NSURL URLWithString:@"http://www.baidu.com/search?id=1"];
NSLog(@"scheme:%@", [url scheme]); //协议 http
NSLog(@"host:%@", [url host]);     //域名 www.baidu.com
NSLog(@"absoluteString:%@", [url absoluteString]); //完整的url字符串 http://www.baidu.com:8080/search?id=1   (刚才在真机上跑了一下，并没有打印出来端口 8080 啊)

NSLog(@"port :%@", [url port]);  // 端口 8080

NSLog(@"pathComponents:%@", [url pathComponents]); // search
NSLog(@"Query:%@", [url query]);  //参数 id=1
